Future<List> downloadFile(String link, String type) async {
  Directory tempDir = await getTemporaryDirectory();
  String tempFilePath = '${tempDir.path}/downloadedTemp$type';
  await http.Dio().download(link, tempFilePath);
  File file = File(tempFilePath);
  return [file.readAsBytesSync(), file.path];
}
