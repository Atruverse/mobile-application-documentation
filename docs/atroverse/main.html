void main() async {
  runApp(
    /// The whole widget tree is wrapped within ScreenUtilInit
    /// for making use of the features inside flutter_screenutil package.
    ScreenUtilInit(builder: (context, child) {
      return GetMaterialApp(
        debugShowCheckedModeBanner: false,
        /// All the pages are defined the mixin below and are
        /// passed to the <code>getPages</code> parameter to let
        /// GetX know of all the pages inside this app. Since routing is
        /// also done with GetX in this app.
        getPages: AppPages.pages,
        /// App is started with an splash screen then, after the needed
        /// initializations happened, the user will be guided to
        /// either home page or register page.
        home: const SplashScreen(),
        /// All repository and controller binding happen down below.
        initialBinding: GlobalBinding(),
      );
    }),
  );
}
