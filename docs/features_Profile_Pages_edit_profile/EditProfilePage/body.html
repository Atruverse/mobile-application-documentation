@override
Widget body(BuildContext context) {
  return Form(
    key: controller.formKey,
    child: SingleChildScrollView(
      child: Padding(
        padding: AppSpacings.sAll18,
        child: Column(
          children: [
            SizedBox(height: 15.r),
            controller.isLoadingImage.value
                ? Container(
                    height: 150.r,
                    width: 150.r,
                    decoration: const BoxDecoration(shape: BoxShape.circle),
                    child: const GlobalLoadingWidget(),
                  )
                : GestureDetector(
                    onTap: controller.changeImageDialog,
                    child: Stack(
                      children: [
                        Container(
                          height: 150.r,
                          width: 150.r,
                          margin: AppSpacings.sAll10,
                          decoration: BoxDecoration(
                            border: Border.all(
                              color: AppColors.grey700,
                              width: 1.25,
                            ),
                            color: AppColors.primary.withOpacity(.35),
                            shape: BoxShape.circle,
                          ),
                          child: controller.imageBytes != null
                              ? ClipRRect(
                                  borderRadius: BorderRadius.circular(125.r),
                                  child: Image.memory(
                                    controller.imageBytes!,
                                    height: 125.r,
                                    width: 125.r,
                                    fit: BoxFit.cover,
                                  ),
                                )
                              : Icon(
                                  CupertinoIcons.person_alt,
                                  size: 100.r,
                                ),
                        ),
                        Positioned(
                          bottom: 10.r,
                          right: 10.r,
                          child: const CircleAvatar(
                            child: Icon(CupertinoIcons.camera),
                          ),
                        )
                      ],
                    ),
                  ),
            SizedBox(height: 15.r),
            Row(
              children: [
                Expanded(
                  child: SelectGenderBox(
                    currentIndex: controller.currentGenderIndex.value,
                    boxIndex: 0,
                    title: 'MALE',
                    changeFunction: () {
                      controller.changeGender(0);
                    },
                  ),
                ),
                SizedBox(width: 3.r),
                Expanded(
                  child: SelectGenderBox(
                    currentIndex: controller.currentGenderIndex.value,
                    boxIndex: 1,
                    title: 'FEMALE',
                    changeFunction: () {
                      controller.changeGender(1);
                    },
                  ),
                ),
                SizedBox(width: 3.r),
                Expanded(
                  child: SelectGenderBox(
                    currentIndex: controller.currentGenderIndex.value,
                    boxIndex: 2,
                    title: 'OTHER',
                    changeFunction: () {
                      controller.changeGender(2);
                    },
                  ),
                ),
              ],
            ),
            const SizedBox(height: 20),
            GlobalInputBox(
              label: 'Bio',
              controller: controller.bioCtrl,
              minLines: 3,
              maxLines: 3,
              validator: (value) {
                if (value!.isEmpty) {
                  return "Enter Your Bio";
                }
              },
            ),
            SizedBox(height: 10.r),
            GlobalInputBox(
              label: 'Name',
              controller: controller.fNameCtrl,
              validator: (value) {
                if (value!.isEmpty) {
                  return "Enter Your Name";
                }
              },
            ),
            SizedBox(height: 10.r),
            GlobalInputBox(
              label: 'Family name',
              controller: controller.lNameCtrl,
              validator: (value) {
                if (value!.isEmpty) {
                  return "Enter Your Family Name";
                }
              },
            ),
            SizedBox(height: 10.r),
            SizedBox(
              width: Get.width,
              child: Row(
                children: [
                  Expanded(
                    child: GlobalInputBox(
                      label: 'User name',
                      controller: controller.usernameCtrl,
                      validator: (value) {
                        if (value!.isEmpty) {
                          return "Enter Your Username";
                        }
                        if (value.length < 5) {
                          return 'Enter at least 5 characters!';
                        }
                      },
                      inputFormatters: <TextInputFormatter>[
                        FilteringTextInputFormatter.allow(
                            RegExp('[a-z A-Z 0-9 . _]')),
                        FilteringTextInputFormatter.deny(RegExp(r'\s')),
                      ],
                      changeFunction: (String value) {
                        controller.changeUsernameStatus(false, false);
                        if (value.length < 4) {
                          return;
                        }
                        try {
                          controller.checkUsernameTimer!.cancel();
                        } catch (_) {}
                        int start = 2;
                        controller.checkUsernameTimer = Timer.periodic(
                            const Duration(seconds: 1), (timer) {
                          if (start == 0) {
                            timer.cancel();
                            controller.checkUsername();
                          } else {
                            start--;
                          }
                        });
                      },
                    ),
                  ),
                  controller.showSuccessUsername.value
                      ? Container(
                          padding: AppSpacings.sAll4,
                          child: const Icon(
                            Icons.check,
                            color: AppColors.primary,
                          ),
                        )
                      : controller.showErrorUsername.value
                          ? Container(
                              padding: AppSpacings.sAll4,
                              child: const Icon(
                                Icons.error,
                                color: Color(0xFFFA0000),
                              ),
                            )
                          : Container(),
                ],
              ),
            ),
            SizedBox(height: 10.r),
            GlobalDatePicker(
              label: 'Birthday',
              controller: controller.birthdayCtrl,
              dateLabelCtrl: controller.birthdayCtrl,
              initialDate: controller.birthday,
              maxDate: DateTime(DateTime.now().year - 13),
            ),
            SizedBox(height: 10.r),
            CountryStateCityPicker(
              country: controller.countryCtrl,
              state: controller.stateCtrl,
              city: controller.cityCtrl,
            ),
            SizedBox(height: 10.r),

            // SelectState(
            //   onCountryChanged: (value) {},
            //   onStateChanged: (value) {},
            //   onCityChanged: (value) {},
            // ),
            GlobalSubmitButton(
              func: controller.editProfile,
              title: 'Submit',
            ),
            SizedBox(height: 50.r),
          ],
        ),
      ),
    ),
  );
}
