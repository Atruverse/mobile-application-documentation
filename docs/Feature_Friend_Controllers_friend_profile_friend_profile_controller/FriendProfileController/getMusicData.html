getMusicData({bool refresh = false}) async {
  if (refresh) {
    currentPageMusic = 1;
    isPageLocked = false;
    dataMusic.clear();
  }
  if (isPageLocked) {
    return;
  }
  if (currentPageMusic == 1) {
    isPageLoading.value = true;
  } else {
    isPageLoadingMore.value = true;
  }
  var result =
      await _repo.friendMusicList(page: currentPageMusic, id: friendId);
  if (result.resultData != null) {
    if (!result.haveNextPage!) {
      isPageLocked = true;
    }
    if (result.resultData!.isEmpty) {
      isPageLocked = true;
    } else {
      dataMusic.addAll(result.resultData!);
    }
    currentPageMusic += 1;
  }
  isPageLoading.value = false;
  isPageLoadingMore.value = false;
}
